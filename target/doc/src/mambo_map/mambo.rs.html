92impl PartialEq for WhatIdo {
93    fn eq(&self, other: &Self) -> bool {
94        match (self, other) {
95            (WhatIdo::INSERT, WhatIdo::INSERT) => true,

109    pub fn new(num_shards: usize, redream_factor: f32) -> Result<Self, &'static str> {
110        if !(1.0..11.0).contains(&redream_factor) {
111            return Err("(1.0..11.0).contains( redream_factor)");
112        }
113        let shards = (0..num_shards)
114            .map(|_| {
115                let data: Box<[Vecta<T>]> = (0..PRIME_NUMBERS_TO_TAB[0])
116                    .map(|_| Mutex::new((ONPLACE, vec![])))
117                    .collect();
118
119                (
120                    Mutex::new(0_u32),
121                    Mutex::new(0_usize),
122                    RwLock::new([data, vec![].into_boxed_slice()]),
123                )
124            })
125            .collect();
126
127        Ok(Self {
128            data_arc: Arc::new((redream_factor, shards)),
129            how_edit_elem_without_update: vec![0; num_shards].into_boxed_slice(),
130        })
131    }

133    fn resize_shard(&self, shard_index: usize, new_len: usize) -> Result<bool, &'static str> {
134        let shard = self
135            .data_arc
136            .1
137            .get(shard_index)
138            .ok_or("shard_index out of range")?;
139        /*locker_resizer is needed for a global announcement that the size of the current shard is currently
140        being edited. if locker_resizer is currently blocked, then other functions donot need to resize this shard.
141        !NOTE!
142        some compilers with aggressive optimization parameters may see that the data in locker_resizer
143        is not being used and may remove the lock call to locker_resizer, which will lead to UB.
144        therefore, garbage is written to locker_resizer, which does not affect the operation of the program.*/
145        let mut locker_resizer = match shard.0.try_lock() {
146            Ok(guard) => guard,
147            Err(TryLockError::WouldBlock) => {
148                return Ok(false);
149            }
150            Err(TryLockError::Poisoned(err)) => {
151                panic!("Mutex poisoned: {:?}", err);
152            }
153        };
154        *locker_resizer = locker_resizer.wrapping_add(1);
155
156        let _ = {
157            //non lock read
158            /*
159            checking that the vector number 0 has a length, which means that it is initialized,
160            it must be of non-zero length. the vector number 1 has a length of 0. If it has a length of 0,
161            it means that an error has occurred in the program or the resize_shard
162            function is currently activated in another thread, which is not possible. In real work,
163            this check should always end positively, and is needed to simplify
164            the detection of errors during development.*/
165            let v0_len = shard.2.read().unwrap()[0].len();
166            let v1_len = shard.2.read().unwrap()[1].len();
167
168            if v0_len > 0 && v1_len == 0 {
169                v0_len
170            } else if v1_len > 0 && v0_len > 0 {
171                return Err("vec0 and vec1 > 0 ");
172            } else {
173                return Err("vec0 and vec1 == 0 ");
174            }
175        };
176
177        {
178            //a new vector is created, with a length of PRIME_NUMBERS_TO_TAB[new_index_in_prime]
179            let new_vec_box: Box<[Vecta<T>]> = (0..new_len)
180                .map(|_| Mutex::new((ONPLACE, vec![])))
181                .collect();
182            //write lock edits putting a new vector in the Rwlock
183            let mut w_rlock = shard.2.write().unwrap();
184            if w_rlock[1].len() == 0 {
185                w_rlock[1] = new_vec_box;
186            } else {
187                return Err("vec 1 is resizes in this moment");
188            }
189            //println!("vec in rw 1len {}", w_rlock[0].len());
190        }
191        {
192            //read non lock
193            let r_lock = shard.2.read().unwrap();
194            let v0_old = &r_lock[0];
195            let v1_new = &r_lock[1];
196            //iterating through the elements in the old vector and moving the elements to the new one
197            for x in v0_old.iter() {
198                let mut temp_old_mutexer = x.lock().unwrap();
199                /*if the elements were moved before they were moved, an error is triggered.
200                since the elements are moved only during copying, and if they have been moved and some other
201                function is also moving them at the moment, this should not be possible,
202                since locker_resizer must ensure that only one fn resize_shard is active at
203                one time for the current shard.*/
204                if temp_old_mutexer.0 == WASMOVED {
205                    return Err(
206                        "Unknown error element that should not be moved, was moved before moving",
207                    );
208                }
209                temp_old_mutexer.0 = WASMOVED;
210
211                for old_elem in temp_old_mutexer.1.iter() {
212                    let mut new_mutex_elem =
213                        v1_new[old_elem.1 as usize % v1_new.len()].lock().unwrap();
214
215                    new_mutex_elem.1.push(*old_elem);
216                    //self.insert_in_mutex(obje_old, &mut new_mutex_elem.1)?;
217                }
218                //minimizing memory consumption, removing unused vectors
219                if temp_old_mutexer.1.capacity() > 0 {
220                    temp_old_mutexer.1 = Vec::with_capacity(0);
221                }
222
223                //trash changes
224                *locker_resizer = locker_resizer.wrapping_add(temp_old_mutexer.1.len() as u32);
225            }
226        }
227        {
228            //write lock edits
229            let mut w_lock = shard.2.write().unwrap();
230
231            let old_1 = mem::replace(&mut w_lock[1], Vec::new().into_boxed_slice());
232            w_lock[0] = old_1;
233            #[cfg(test)]
234            {
235                //  println!("resize_shard {}  len {}", shard_index, w_lock[0].len())
236            }
237        }
238
239        Ok(true)
240    }

242    #[inline(always)]
243    fn search_vec<F>(&mut self, index: usize, operation: F) -> Result<(), &'static str>
244    where
245        F: FnOnce(&mut Vec<ElemBox<T>>) -> Result<WhatIdo, &'static str>,
246    {
247        let mut is_edit = false;
248        let mut shard_capasity = 0;
249        let shard_index = index % self.data_arc.1.len();
250        let shard = &self.data_arc.1[shard_index];
251
252        {
253            let r_lock = shard.2.read().unwrap();
254            for x in 0..2 {
255                let vecta = &r_lock[x & 0b1];
256
257                if 0 != vecta.len() {
258                    let mutexer = &mut vecta[index % vecta.len()].lock().unwrap();
259                    if ONPLACE == mutexer.0 {
260                        shard_capasity = vecta.len();
261
262                        match operation(&mut mutexer.1)? {
263                            WhatIdo::REMOVE => {
264                                self.how_edit_elem_without_update[shard_index] = self
265                                    .how_edit_elem_without_update[shard_index]
266                                    .checked_sub(1)
267                                    .ok_or("how_edit_elem_without_update sub is owerflow")?;
268                                is_edit = true;
269                            }
270                            WhatIdo::INSERT => {
271                                self.how_edit_elem_without_update[shard_index] = self
272                                    .how_edit_elem_without_update[shard_index]
273                                    .checked_add(1)
274                                    .ok_or("how_edit_elem_without_update add is owerflow")?;
275                                is_edit = true;
276                            }
277                            _ => {
278                                return Ok(());
279                            }
280                        };
281                        break;
282                    }
283                }
284            }
285        } //unlock read r_lock
286        if is_edit {
287            // println!("shard_capasity: {}", shard_capasity);
288            if let Some(new_len) = self.new_size_shard(shard_index, shard_capasity) {
289                // println!("is_edit: {} {}", new_len, shard_capasity);
290                self.resize_shard(index % self.data_arc.1.len(), new_len)?;
291            }
292            return Ok(());
293        }
294
295        Err("access error, constant employment")
296    }

298    ///Resizing should only be done if the occupancy rate is higher than the redream_factor.
299    /// if it is higher and these are not the last elements in PRIME_NUMBERS_TO_TAB,
300    /// then its size becomes PRIME_NUMBERS_TO_TAB[+1],
301    /// if the number of elements is so large that you can use PRIME_NUMBERS_TO_TAB[-1]
302    ///  and there will still be room under 25% before exceeding edream_factor,
303    ///  the capacity size decreases by PRIME_NUMBERS_TO_TAB[-1].
304    #[inline(always)]
305    fn new_size_shard(&mut self, shard_index: usize, shard_capasity: usize) -> Option<usize> {
306        let shard_i = &mut self.how_edit_elem_without_update[shard_index];
307
308        if shard_i.abs_diff(0) < ELEM_NUMS_TO_READ_IN_MUTEX {
309            return None;
310        }
311
312        let elems_in_me = {
313            let mut mutexer = self.data_arc.1[shard_index].1.lock().unwrap();
314
315            if *shard_i > 0 {
316                *mutexer = mutexer.checked_add(shard_i.abs_diff(0) as usize).unwrap();
317            } else {
318                *mutexer = mutexer.checked_sub(shard_i.abs_diff(0) as usize).unwrap();
319            }
320            *shard_i = 0;
321            *mutexer
322        };
323        //println!("elems in me: {}", elems_in_me);
324        let index_my_prime = self.difer_index(shard_capasity);
325
326        if elems_in_me as f32 / (shard_capasity as f32 + EPSILON) > self.data_arc.0 {
327            if index_my_prime + 1 < PRIME_NUMBERS_TO_TAB.len() {
328                Some(PRIME_NUMBERS_TO_TAB[index_my_prime + 1] as usize)
329            } else {
330                None
331            }
332        } else if index_my_prime > 0 {
333            let t = PRIME_NUMBERS_TO_TAB[index_my_prime - 1];
334            if self.data_arc.0 > (elems_in_me as f32 * SHRINK_THRESHOLD_FACTOR) / t as f32 {
335                Some(PRIME_NUMBERS_TO_TAB[index_my_prime - 1] as usize)
336            } else {
337                None
338            }
339        } else {
340            None
341        }
342    }

344    fn difer_index(&self, target: usize) -> usize {
345        let target = target as u64;
346
347        PRIME_NUMBERS_TO_TAB
348            .iter()
349            .enumerate()
350            .min_by_key(|&(_, &prime)| prime.abs_diff(target))
351            .map(|(index, _)| index)
352            .unwrap_or(0)
353    }

362    pub fn insert(&mut self, index: usize, elem: T) -> Result<(), &'static str> {
363        self.search_vec(index, |vecta| {
364            for x in vecta.iter() {
365                if x.1 == index as u64 {
366                    return Err("there is already an element");
367                }
368            }
369            vecta.push((elem, index as u64));
370
371            Ok(WhatIdo::INSERT)
372        })
373    }

375    pub fn remove(&mut self, index: usize, none_is_err: bool) -> Result<Option<T>, &'static str> {
376        let mut ret_after_remove: Option<T> = None;
377
378        self.search_vec(index, |vecta| {
379            for x in 0..vecta.len() {
380                if vecta[x].1 == index as u64 {
381                    let last = *vecta.last().ok_or("ver item is empty")?;
382
383                    ret_after_remove = Some(vecta[x].0.clone());
384                    vecta[x] = last;
385                    vecta.pop();
386                    return Ok(WhatIdo::REMOVE);
387                }
388            }
389            if none_is_err {
390                Err("elem not in map")
391            } else {
392                Ok(WhatIdo::READ)
393            }
394        })?;
395        Ok(ret_after_remove)
396    }

398    pub fn read<RFy>(&mut self, index: usize, ridler: RFy) -> Result<(), &'static str>
399    where
400        RFy: FnOnce(&mut T) -> Result<(), &'static str>,
401    {
402        self.search_vec(index, |vecta| {
403            for (t, hahs) in vecta.iter_mut() {
404                if *hahs == index as u64 {
405                    ridler(t)?;
406                    return Ok(WhatIdo::READ);
407                }
408            }
409            Err("element not in map")
410        })
411    }

413    pub fn elems_im_me(&self) -> Result<usize, &'static str> {
414        let mut elems = 0_usize;
415        for x in self.data_arc.1.iter() {
416            let t = x.1.lock().unwrap();
417            elems = elems
418                .checked_add(*t)
419                .ok_or("err elems.checked_add(elems in shard )")?;
420        }
421        Ok(elems)
422    }
